}; $objWriter = new \PHPExcel_Writer_Excel5($objPHPExcel);//保存excel—2007格式 $name = 'delivery' . date('YmdHis'); header('Content-Type: application/vnd.ms-excel'); header('Content-Disposition: attachment;filename="' . $name . '.xls"'); header('Cache-Control: max-age=0'); $objWriter->save('php://output');
